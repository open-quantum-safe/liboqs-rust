27macro_rules! implement_kems {
28    { $(($feat: literal) $kem: ident: $oqs_id: ident),* $(,)? } => (
29
30        /// Supported algorithms by OQS
31        ///
32        /// Note that this doesn't mean that they'll be available.
33        ///
34        /// Optional support for `serde` if that feature is enabled.
35        #[derive(Clone, Copy, Debug, Eq, PartialEq, Hash)]
36        #[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
37        #[allow(missing_docs)]
38        pub enum Algorithm {
39            $(
40                $kem,
41            )*
42        }
43
44        fn algorithm_to_id(algorithm: Algorithm) -> *const libc::c_char {
45            let id: &[u8] = match algorithm {
46                $(
47                    Algorithm::$kem => &ffi::$oqs_id[..],
48                )*
49            };
50            id as *const _ as *const libc::c_char
51        }
52
53        $(
54            #[cfg(test)]
55            #[allow(non_snake_case)]
56            mod $kem {
57                use super::*;
58
59                #[test]
60                #[cfg(feature = $feat)]
61                fn test_encaps_decaps() -> Result<()> {
62                    crate::init();
63
64                    let alg = Algorithm::$kem;
65                    let kem = Kem::new(alg)?;
66                    let (pk, sk) = kem.keypair()?;
67                    let (ct, ss1) = kem.encapsulate(&pk)?;
68                    let ss2 = kem.decapsulate(&sk, &ct)?;
69                    assert_eq!(ss1, ss2, "shared secret not equal!");
70                    Ok(())
71                }
72
73                #[test]
74                #[cfg(feature = $feat)]
75                fn test_encaps_decaps_derand() -> Result<()> {
76                    use crate::ffi::rand::OQS_randombytes;
77                    crate::init();
78
79                    let alg = Algorithm::$kem;
80                    let kem = Kem::new(alg)?;
81                    let mut seed = KeypairSeed {
82                        bytes: Vec::with_capacity(kem.length_keypair_seed()),
83                    };
84                    unsafe {
85                        // On some systems, getentropy fails if given a zero-length array
86                        if (kem.length_keypair_seed() > 0) {
87                            OQS_randombytes(seed.bytes.as_mut_ptr(), kem.length_keypair_seed());
88                        }
89                        seed.bytes.set_len(kem.length_keypair_seed());
90                    }
91                    let result = kem.keypair_derand(&seed);
92                    // expect Error::Error for KEMs with this API disabled
93                    if (kem.length_keypair_seed() == 0) {
94                        return result.map_or_else(|e| { match e { Error::Error => Ok(()), _ => Err(Error::Error) } }, |_| Err(Error::Error));
95                    }
96                    let (pk, sk) = result?;
97                    let (ct, ss1) = kem.encapsulate(&pk)?;
98                    let ss2 = kem.decapsulate(&sk, &ct)?;
99                    assert_eq!(ss1, ss2, "shared secret not equal!");
100                    Ok(())
101                }
102
103                #[test]
104                fn test_enabled() {
105                    crate::init();
106                    if cfg!(feature = $feat) {
107                        assert!(Algorithm::$kem.is_enabled());
108                    } else {
109                        assert!(!Algorithm::$kem.is_enabled())
110                    }
111                }
112
113                #[test]
114                fn test_name() {
115                    let algo = Algorithm::$kem;
116                    // Just make sure the name impl does not panic or crash.
117                    let name = algo.name();
118                    #[cfg(feature = "std")]
119                    assert_eq!(name, algo.to_string());
120                    // ... And actually contains something.
121                    assert!(!name.is_empty());
122                }
123
124                #[test]
125                fn test_get_algorithm_back() {
126                    let algorithm = Algorithm::$kem;
127                    if algorithm.is_enabled() {
128                        let kem = Kem::new(algorithm).unwrap();
129                        assert_eq!(algorithm, kem.algorithm());
130                    }
131                }
132
133                #[test]
134                fn test_version() {
135                    if let Ok(kem) = Kem::new(Algorithm::$kem) {
136                        // Just make sure the version can be called without panic
137                        let version = kem.version();
138                        // ... And actually contains something.
139                        assert!(!version.is_empty());
140                    }
141                }
142            }
143        )*
144    )
145}

179impl Algorithm {
180    /// Returns true if this algorithm is enabled in the linked version
181    /// of liboqs
182    pub fn is_enabled(self) -> bool {
183        unsafe { ffi::OQS_KEM_alg_is_enabled(algorithm_to_id(self)) == 1 }
184    }
185
186    /// Provides a pointer to the id of the algorithm
187    ///
188    /// For use with the FFI api methods
189    pub fn to_id(self) -> *const libc::c_char {
190        algorithm_to_id(self)
191    }
192
193    /// Returns the algorithm's name as a static Rust string.
194    ///
195    /// This is the same as the `to_id`, but as a safe Rust string.
196    pub fn name(&self) -> &'static str {
197        // SAFETY: The id from ffi must be a proper null terminated C string
198        let id = unsafe { CStr::from_ptr(self.to_id()) };
199        id.to_str().expect("OQS algorithm names must be UTF-8")
200    }
201}
202
203#[cfg(feature = "std")]
204impl std::fmt::Display for Algorithm {
205    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
206        self.name().fmt(f)
207    }
208}

210/// KEM algorithm
211///
212/// # Example
213/// ```rust
214/// # if !cfg!(feature = "ml_kem") { return; }
215/// use oqs;
216/// oqs::init();
217/// let kem = oqs::kem::Kem::new(oqs::kem::Algorithm::MlKem512).unwrap();
218/// let (pk, sk) = kem.keypair().unwrap();
219/// let (ct, ss) = kem.encapsulate(&pk).unwrap();
220/// let ss2 = kem.decapsulate(&sk, &ct).unwrap();
221/// assert_eq!(ss, ss2);
222/// ```
223pub struct Kem {
224    algorithm: Algorithm,
225    kem: NonNull<ffi::OQS_KEM>,
226}
227
228unsafe impl Sync for Kem {}
229unsafe impl Send for Kem {}
230
231impl Drop for Kem {
232    fn drop(&mut self) {
233        unsafe { ffi::OQS_KEM_free(self.kem.as_ptr()) };
234    }
235}

247        let kem = unsafe { ffi::OQS_KEM_new(algorithm_to_id(algorithm)) };

249            || Err(Error::AlgorithmDisabled),
250            |kem| Ok(Self { algorithm, kem }),

365    /// Generate a new keypair
366    pub fn keypair(&self) -> Result<(PublicKey, SecretKey)> {
367        let kem = unsafe { self.kem.as_ref() };
368        let func = kem.keypair.unwrap();
369        let mut pk = PublicKey {
370            bytes: Vec::with_capacity(kem.length_public_key),
371        };
372        let mut sk = SecretKey {
373            bytes: Vec::with_capacity(kem.length_secret_key),
374        };
375        let status = unsafe { func(pk.bytes.as_mut_ptr(), sk.bytes.as_mut_ptr()) };
376        status_to_result(status)?;
377        // update the lengths of the vecs
378        // this is safe to do, as we have initialised them now.
379        unsafe {
380            pk.bytes.set_len(kem.length_public_key);
381            sk.bytes.set_len(kem.length_secret_key);
382        }
383        Ok((pk, sk))
384    }

386    /// Generate a new keypair from a seed
387    pub fn keypair_derand<'a, S: Into<KeypairSeedRef<'a>>>(
388        &self,
389        seed: S,
390    ) -> Result<(PublicKey, SecretKey)> {
391        let seed = seed.into();
392        if seed.bytes.len() != self.length_keypair_seed() {
393            return Err(Error::InvalidLength);
394        }
395        let kem = unsafe { self.kem.as_ref() };
396        let func = kem.keypair_derand.unwrap();
397        let mut pk = PublicKey {
398            bytes: Vec::with_capacity(kem.length_public_key),
399        };
400        let mut sk = SecretKey {
401            bytes: Vec::with_capacity(kem.length_secret_key),
402        };
403        let status = unsafe {
404            func(
405                pk.bytes.as_mut_ptr(),
406                sk.bytes.as_mut_ptr(),
407                seed.bytes.as_ptr(),
408            )
409        };
410        status_to_result(status)?;
411        // update the lengths of the vecs
412        // this is safe to do, as we have initialised them now.
413        unsafe {
414            pk.bytes.set_len(kem.length_public_key);
415            sk.bytes.set_len(kem.length_secret_key);
416        }
417        Ok((pk, sk))
418    }

420    /// Encapsulate to the provided public key
421    pub fn encapsulate<'a, P: Into<PublicKeyRef<'a>>>(
422        &self,
423        pk: P,
424    ) -> Result<(Ciphertext, SharedSecret)> {
425        let pk = pk.into();
426        if pk.bytes.len() != self.length_public_key() {
427            return Err(Error::InvalidLength);
428        }
429        let kem = unsafe { self.kem.as_ref() };
430        let func = kem.encaps.unwrap();
431        let mut ct = Ciphertext {
432            bytes: Vec::with_capacity(kem.length_ciphertext),
433        };
434        let mut ss = SharedSecret {
435            bytes: Vec::with_capacity(kem.length_shared_secret),
436        };
437        // call encapsulate
438        let status = unsafe {
439            func(
440                ct.bytes.as_mut_ptr(),
441                ss.bytes.as_mut_ptr(),
442                pk.bytes.as_ptr(),
443            )
444        };
445        status_to_result(status)?;
446        // update the lengths of the vecs
447        // this is safe to do, as we have initialised them now.
448        unsafe {
449            ct.bytes.set_len(kem.length_ciphertext);
450            ss.bytes.set_len(kem.length_shared_secret);
451        }
452        Ok((ct, ss))
453    }
454
455    /// Decapsulate the provided ciphertext
456    pub fn decapsulate<'a, 'b, S: Into<SecretKeyRef<'a>>, C: Into<CiphertextRef<'b>>>(
457        &self,
458        sk: S,
459        ct: C,
460    ) -> Result<SharedSecret> {
461        let kem = unsafe { self.kem.as_ref() };
462        let sk = sk.into();
463        let ct = ct.into();
464        if sk.bytes.len() != self.length_secret_key() || ct.bytes.len() != self.length_ciphertext()
465        {
466            return Err(Error::InvalidLength);
467        }
468        let mut ss = SharedSecret {
469            bytes: Vec::with_capacity(kem.length_shared_secret),
470        };
471        let func = kem.decaps.unwrap();
472        // Call decapsulate
473        let status = unsafe { func(ss.bytes.as_mut_ptr(), ct.bytes.as_ptr(), sk.bytes.as_ptr()) };
474        status_to_result(status)?;
475        // update the lengths of the vecs
476        // this is safe to do, as we have initialised them now.
477        unsafe { ss.bytes.set_len(kem.length_shared_secret) };
478        Ok(ss)
479    }
