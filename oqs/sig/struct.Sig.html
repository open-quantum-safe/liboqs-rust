Signature scheme

§Example

use oqs;
oqs::init();
let scheme = oqs::sig::Sig::new(oqs::sig::Algorithm::MlDsa44).unwrap();
let message = [0u8; 100];
let (pk, sk) = scheme.keypair().unwrap();
let signature = scheme.sign(&message, &sk).unwrap();
assert!(scheme.verify(&message, &signature, &pk).is_ok());
