KEM algorithm

ยงExample

use oqs;
oqs::init();
let kem = oqs::kem::Kem::new(oqs::kem::Algorithm::MlKem512).unwrap();
let (pk, sk) = kem.keypair().unwrap();
let (ct, ss) = kem.encapsulate(&pk).unwrap();
let ss2 = kem.decapsulate(&sk, &ct).unwrap();
assert_eq!(ss, ss2);
